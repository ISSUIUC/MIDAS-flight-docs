18IMU IMUSensor::read(){

20
21 IMU reading{};
22

26 &reading.lowg_acceleration.ay,

34 &reading.highg_acceleration.az);

36

42 }

147void IMUSensor::next_calib(BuzzerController& buzzer, EEPROMController& eeprom) {
148 IMUCalibrationState next_state = static_cast<IMUCalibrationState>(static_cast<int>(calibration_state) + 1);
149 if (next_state == IMUCalibrationState::CALIB_DONE) {
150 // Calibration is finished, we will commit to EEPROM
151 calibration_state = IMUCalibrationState::NONE;
152 eeprom.data.lsm6dsv320x_hg_xl_bias = calibration_sensor_bias;
153 eeprom.commit();
154 buzzer.play_tune(xl_calib_done, C_XL_LENGTH);
155 } else {
156 calibration_state = next_state;
157 buzzer.play_tune(xl_calib_next_axis, C_XL_LENGTH);
158 }
159 _calib_valid_readings = 0;
160}
