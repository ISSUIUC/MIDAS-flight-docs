28// ---- Kamaji Thread ----
29DECLARE_THREAD(hilsim, RocketSystems* arg) {
30 uint8_t header[5];
31 uint8_t crc_buf[2];
32 uint32_t timestamp;
33 uint8_t discriminant;
34 size_t data_size;
35 uint8_t data_buf[128];
36 uint16_t crc;
37
38 sys_flags_t sflags;
39
40 while (true) {
41 int delim;
42 do {
43 k_tick();
44 if (Serial.available() == 0) {
45 THREAD_SLEEP(1);
46 }
47 delim = Serial.read();
48
49 if (sflags.fsm_target != FSMState::FSM_STATE_COUNT) {
50 arg->rocket_data.fsm_state.update({sflags.fsm_target, sflags.fsm_target_motor});
51 sflags.fsm_target = FSMState::FSM_STATE_COUNT;
52 }
53
54 } while(delim != '$' && delim != '#' && delim != '&');
55
56 if(delim == '&') {
57 START_THREAD(shell, DATA_CORE, arg, 5);
58 vTaskDelete(NULL);
59 continue;
60 }
61
62 if(delim == '#') {
63 if (!k_read_exact(data_buf, 16, 10)) continue;
64
65 // Sys message is 14 byte data, 2 byte crc
66 memcpy(&crc, data_buf + 14, sizeof(crc_buf));
67 k_handle_sys_msg(data_buf, crc, &sflags);
68 continue;
69 }
70
71 if(delim == '$') {
72 if(!k_read_exact(header, sizeof(header), 10)) continue;
73
74 memcpy(&timestamp, &header[0], 4);
75 discriminant = header[4];
76 data_size = k_get_discriminant_size(discriminant);
77
78 if (!k_read_exact(data_buf, (size_t)data_size, 10)) continue;
79 if (!k_read_exact(crc_buf, sizeof(crc_buf), 5)) continue;
80
81 memcpy(&crc, crc_buf, sizeof(crc_buf));
82 k_handle_reading(timestamp, discriminant, data_buf, data_size, crc);
83
84 }
85
86 k_tick();
87 THREAD_SLEEP(1);
88 }
89}
