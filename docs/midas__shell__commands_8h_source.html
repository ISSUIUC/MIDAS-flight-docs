525// Returns true if the path refers to an active logfile. Compares basenames so
526// any number of leading slashes is tolerated.
527static bool is_active_log(RocketSystems* arg, const char* path) {
528 const char* basename = strrchr(path, '/');
529 basename = basename ? basename + 1 : path;
530
531 char bin[20];
532 char meta[20];
533 snprintf(bin, sizeof(bin), "data%u.bin", arg->log_sink.current_file_no - 1);
534 snprintf(meta, sizeof(meta), "data%u.meta", arg->log_sink.current_file_no - 1);
535 return !strcmp(basename, bin) || !strcmp(basename, meta);
536}

538bool delete_all_flash(RocketSystems* arg) {
539 File root = SD_MMC.open("/");
540 while (true) {
541 File file = root.openNextFile();
542 if (!file) {
543 break; // No more files
544 }
545 String fullpath = "/" + String(file.name());
546 file.close();
547 file.flush();
548 if (is_active_log(arg, fullpath.c_str())) {
549 Serial.print("Skipping active log: ");
550 Serial.println(fullpath);
551 continue;
552 }
553 Serial.print("Deleting file: ");
554 Serial.println(fullpath);
555 if(!SD_MMC.remove(fullpath.c_str())) {
556 Serial.print("Failed to delete ");
557 Serial.println(fullpath);
558 return false;
559 }
560 }
561 return true;
562}

564MCommandExecutionResult cmd_lfd(const MShellContext& ctx) {
565 // Launch File Dump: Outputs a .launch file as binary data over serial
566
567 // The new .launch format is as follows:
568 // LAUNCH <version_no>
569 // FILE <original_filename>
570 // CHECKSUM <log_checksum> <eeprom_checksum>
571 // META <meta file size>
572 // BIN <bin file size>
573 // <meta file content>
574 // <bin file content>
575
576 if (ctx.argc != 2 && ctx.argc != 3) { }
577 bool stop_after_header_print = false;
578 if(ctx.argc == 3) {
579 if(!strcmp(ctx.argv[2], "--head")) {
580 stop_after_header_print = true;
581 } else {
582 return MCommandExecutionResult::ERR_INVAL_ARGUMENT;
583 }
584 } else if (ctx.argc != 2) {
585 return MCommandExecutionResult::ERR_INVAL_ARGC;
586 }
587
588
589 String bin_path = "/" + String(ctx.argv[1]) + ".bin";
590 String meta_path = "/" + String(ctx.argv[1]) + ".meta";
591 File bin_file = SD_MMC.open(bin_path.c_str(), FILE_READ);
592 File meta_file = SD_MMC.open(meta_path.c_str(), FILE_READ);
593
594 if(!bin_file) {
595 Serial.println("Failed to open .bin file!");
596 return MCommandExecutionResult::ERR_FS_FAIL_OPEN;
597 }
598
599 if(!meta_file) {
600 Serial.println("Failed to open .meta file!");
601 return MCommandExecutionResult::ERR_FS_FAIL_OPEN;
602 }
603
604 size_t size_ptr = LOG_FMT_VERSION; // We write multiple uint32_ts, so for serial.write to print it properly we need to store it
605
606 // Start writing header
607 Serial.write("LAUNCH "); Serial.write((uint8_t*)&size_ptr, sizeof(size_t)); Serial.write('\n');
608
609 // File name
610 Serial.write("FILE "); Serial.write(ctx.argv[1]); Serial.write('\n');
611
612 // Log checksum
613 size_ptr = LOG_CHECKSUM;
614 Serial.write("CHECKSUM "); Serial.write((uint8_t*)&size_ptr, sizeof(size_t)); Serial.write(" ");
615
616 // EEPROM checksum
617 size_ptr = EEPROM_CHECKSUM;
618 Serial.write((uint8_t*)&size_ptr, sizeof(size_t)); Serial.write('\n');
619
620 // Meta file size
621 size_ptr = meta_file.size();
622 Serial.write("META "); Serial.write((uint8_t*)&size_ptr, sizeof(size_t)); Serial.write('\n');
623
624 // Binary file size
625 size_ptr = bin_file.size();
626 Serial.write("BIN "); Serial.write((uint8_t*)&size_ptr, sizeof(size_t)); Serial.write('\n');
627
628 Serial.flush();
629
630 if(stop_after_header_print) { return MCommandExecutionResult::OK; }
631
632 // Then dump meta and bin files
633 while(meta_file.available()) {
634 Serial.write(meta_file.read());
635 }
636 meta_file.close();
637
638 while(bin_file.available()) {
639 Serial.write(bin_file.read());
640 }
641 bin_file.close();
642
643 Serial.flush();
644
645 return MCommandExecutionResult::OK;
646}
