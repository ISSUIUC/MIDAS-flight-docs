142void Yessir::update(Barometer barometer, Acceleration acceleration, Orientation orientation, FSMState FSM_state) {
143 if (FSM_state == FSMState::STATE_FIRST_BOOST || FSM_state == FSMState::STATE_SECOND_BOOST) {
144 float sum = 0;
145 float data[10];
146 alt_buffer.readSlice(data, 0, 10);
147 for (float i : data) {
148 sum += i;
149 }
150 KalmanState kalman_state = (KalmanState){sum / 10.0f, 0, 0, 0, 0, 0, 0, 0, 0};
151 setState(kalman_state);
152 } else if (FSM_state >= FSMState::STATE_APOGEE) {
153 H(1, 2) = 0;
154 }
155
156 Eigen::Matrix<float, 4, 4> S_k = Eigen::Matrix<float, 4, 4>::Zero();
157 S_k = (((H * P_priori * H.transpose()) + R)).inverse();
158 Eigen::Matrix<float, 9, 9> identity = Eigen::Matrix<float, 9, 9>::Identity();
159 K = (P_priori * H.transpose()) * S_k;
160
161 // Sensor Measurements
162 Eigen::Matrix<float, 3, 1> accel = Eigen::Matrix<float, 3, 1>::Zero();
163
164 accel(0, 0) = acceleration.az - 0.045;
165 accel(1, 0) = acceleration.ay - 0.065;
166 accel(2, 0) = -acceleration.ax - 0.06;
167
168 euler_t angles = orientation.getEuler();
169 angles.yaw = -angles.yaw;
170
171 Eigen::Matrix<float, 3, 1> acc = BodyToGlobal(angles, accel);
172
173 y_k(1, 0) = (acc(0)) * 9.81 - 9.81;
174 y_k(2, 0) = (acc(1)) * 9.81;
175 y_k(3, 0) = (acc(2)) * 9.81;
176
177 y_k(0, 0) = barometer.altitude;
178 alt_buffer.push(barometer.altitude);
179
180
181 // # Posteriori Update
182 x_k = x_priori + K * (y_k - (H * x_priori));
183 P_k = (identity - K * H) * P_priori;
184 // Joseph (Expanded) Form
185 // P_k = (identity - K * H) * P_priori * (identity - K * H).transpose() + K * R * K.transpose();
186
187 kalman_state.state_est_pos_x = x_k(0, 0);
188 kalman_state.state_est_vel_x = x_k(1, 0);
189 kalman_state.state_est_accel_x = x_k(2, 0);
190 kalman_state.state_est_pos_y = x_k(3, 0);
191 kalman_state.state_est_vel_y = x_k(4, 0);
192 kalman_state.state_est_accel_y = x_k(5, 0);
193 kalman_state.state_est_pos_z = x_k(6, 0);
194 kalman_state.state_est_vel_z = x_k(7, 0);
195 kalman_state.state_est_accel_z = x_k(8, 0);
196
197 state.position = (Position){kalman_state.state_est_pos_x,kalman_state.state_est_pos_y,kalman_state.state_est_pos_z};
198 state.velocity = (Velocity){kalman_state.state_est_vel_x,kalman_state.state_est_vel_y,kalman_state.state_est_vel_z};
199 state.acceleration = (Acceleration){kalman_state.state_est_accel_x,kalman_state.state_est_accel_y,kalman_state.state_est_accel_z};
200}

228Eigen::Matrix<float, 3, 1> Yessir::BodyToGlobal(euler_t angles, Eigen::Matrix<float, 3, 1> body_vect) {
229 Eigen::Matrix3f roll, pitch, yaw;
230 roll << 1., 0., 0., 0., cos(angles.roll), -sin(angles.roll), 0., sin(angles.roll), cos(angles.roll);
231 pitch << cos(angles.pitch), 0., sin(angles.pitch), 0., 1., 0., -sin(angles.pitch), 0., cos(angles.pitch);
232 yaw << cos(angles.yaw), -sin(angles.yaw), 0., sin(angles.yaw), cos(angles.yaw), 0., 0., 0., 1.;
233 return yaw * pitch * roll * body_vect;
234}
