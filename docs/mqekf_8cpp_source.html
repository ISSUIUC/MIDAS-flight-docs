108void QuaternionMEKF::time_update(Velocity const &gyro, float Ts)
109{
110 // Conversion from degrees/s to radians/s
111
112 Eigen::Matrix<float, 3, 1> gyr;
113 gyr(0, 0) = gyro.vx * (pi / 180.0f);
114 gyr(1, 0) = gyro.vy * (pi / 180.0f);
115 gyr(2, 0) = gyro.vz * (pi / 180.0f);
116
117 set_transition_matrix(gyr - x.tail(3), Ts);
118
119 Eigen::Vector4f q; // necessary to reorder to w,x,y,z
120 q << qref.w(), qref.x(), qref.y(), qref.z();
121
122 q = F * q;
123
124 qref = Eigen::Quaternionf(q(0), q(1), q(2), q(3));
125 qref.normalize();
126
127 Eigen::Matrix<float, 6, 6> F_a;
128 // Slice 3x3 block from F
129 F_a << F.block(0, 0, 3, 3), (-Eigen::Matrix<float, 3, 3>::Identity() * Ts),
130 Eigen::Matrix<float, 3, 3>::Zero(), Eigen::Matrix<float, 3, 3>::Identity();
131 P = F_a * P * F_a.transpose() + Q; // P update
132}

134void QuaternionMEKF::measurement_update(Acceleration const &accel, Magnetometer const &mag_input)
135{
136 // accel measurements
137 Eigen::Matrix<float, 3, 1> acc;
138 acc(0, 0) = accel.ax;
139 acc(1, 0) = accel.ay;
140 acc(2, 0) = accel.az;
141
142 Eigen::Matrix<float, 3, 1> mag;
143 mag(0, 0) = mag_input.mx;
144 mag(1, 0) = mag_input.my;
145 mag(2, 0) = mag_input.mz;
146
147 Eigen::Matrix<float, 3, 1> const v1hat = accelerometer_measurement_func();
148 Eigen::Matrix<float, 3, 1> const v2hat = magnetometer_measurement_func();
149
150 Eigen::Matrix<float, 3, 3> const C1 = skew_symmetric_matrix(v1hat);
151 Eigen::Matrix<float, 3, 3> const C2 = skew_symmetric_matrix(v2hat);
152
153 Eigen::Matrix<float, 6, 6> C;
154 C << C1, Eigen::Matrix<float, 3, 3>::Zero(),
155 C2, Eigen::Matrix<float, 3, 3>::Zero();
156
157 Eigen::Matrix<float, 6, 1> yhat;
158 yhat << v1hat,
159 v2hat;
160
161 Eigen::Matrix<float, 6, 1> y;
162 y << acc,
163 mag;
164
165 Eigen::Matrix<float, 6, 1> inno = y - yhat;
166
167 Eigen::Matrix<float, 6, 6> const s = C * P * C.transpose() + R;
168
169 // K = P * C.float *(s)^-1
170 // K * s = P*C.float
171
172 // This is the form
173 // x * A = b
174 // Which can be solved with the code below
175 Eigen::FullPivLU<Eigen::Matrix<float, 6, 6>> lu(s); // LU decomposition of s
176 if (lu.isInvertible())
177 {
178 Eigen::Matrix<float, 6, 6> const K = P * C.transpose() * lu.inverse(); // gain
179
180 x += K * inno; // applying correction???
181
182 // Joseph form of covariance measurement update
183 Eigen::Matrix<float, 6, 6> const temp = Eigen::Matrix<float, 6, 6>::Identity() - K * C;
184 P = temp * P * temp.transpose() + K * R * K.transpose(); // covariance update???
185 // Apply correction to qref
186 Eigen::Quaternion<float> corr(1, 0.5f * x(0), 0.5f * x(1), 0.5f * x(2)); // small angle approx????
187 corr.normalize();
188 qref = qref * corr; // multiply quaternions from ref???????
189
190 // We only want to reset the quaternion part of the state
191 x(0) = 0;
192 x(1) = 0;
193 x(2) = 0;
194 }

250void QuaternionMEKF::set_transition_matrix(Eigen::Ref<const Eigen::Matrix<float, 3, 1>> const &gyr, float Ts)
251{
252 Eigen::Matrix<float, 3, 1> const delta_theta = gyr * Ts;
253 float un = delta_theta.norm();
254 if (un == 0)
255 un = std::numeric_limits<float>::min();
256
257 Eigen::Matrix<float, 4, 4> const Omega = (Eigen::Matrix<float, 4, 4>() << -skew_symmetric_matrix(delta_theta), delta_theta,
258 -delta_theta.transpose(), 0)
259 .finished();
260
261 F = std::cos(0.5f * un) * Eigen::Matrix<float, 4, 4>::Identity() + std::sin(0.5f * un) / un * Omega;

292void QuaternionMEKF::initialize_from_acc_mag(Acceleration const &acc_struct, Magnetometer const &mag_struct)
293{
294 Eigen::Matrix<float, 3, 1> acc;
295 acc << acc_struct.ax, acc_struct.ay, acc_struct.az;
296 Eigen::Matrix<float, 3, 1> mag;
297 mag << mag_struct.mx, mag_struct.my, mag_struct.mz;
298 float const anorm = acc.norm();
299 v1ref << anorm, 0, 0;
300
301 Eigen::Matrix<float, 3, 1> const acc_normalized = acc / anorm;
302 Eigen::Matrix<float, 3, 1> const mag_normalized = mag.normalized();
303
304 Eigen::Matrix<float, 3, 1> const Rz = -acc_normalized;
305 Eigen::Matrix<float, 3, 1> const Ry = Rz.cross(mag_normalized).normalized();
306 Eigen::Matrix<float, 3, 1> const Rx = Ry.cross(Rz).normalized();
307
308 // Construct the rotation matrix
309 Eigen::Matrix<float, 3, 3> const R = (Eigen::Matrix<float, 3, 3>() << Rx, Ry, Rz).finished();
310
311 // Eigen can convert it to a quaternion
312 qref = R.transpose();
313
314 // Reference magnetic field vector
315 v2ref = qref * mag;
316}

323Eigen::Matrix<float, 3, 1> QuaternionMEKF::quatToEuler(const Eigen::Matrix<float, 4, 1> &q)
324{
325
326 double w = q[0];
327 double x = q[1];
328 double y = q[2];
329 double z = q[3];
330
331 double sinr_cosp = 2.0 * (w * x + y * z);
332 double cosr_cosp = 1.0 - 2.0 * (x * x + y * y);
333 double roll = std::atan2(sinr_cosp, cosr_cosp);
334
335 double sinp = 2.0 * (w * y - z * x);
336 double pitch;
337 if (std::abs(sinp) >= 1)
338 pitch = std::copysign(pi / 2, sinp);
339 else
340 pitch = std::asin(sinp);
341
342 double siny_cosp = 2.0 * (w * z + x * y);
343 double cosy_cosp = 1.0 - 2.0 * (y * y + z * z);
344 double yaw = std::atan2(siny_cosp, cosy_cosp);
345 return Eigen::Matrix<float, 3, 1>(roll, pitch, yaw);
346}
