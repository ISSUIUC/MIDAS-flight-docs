17static bool k_read_exact(uint8_t* dst, size_t n, unsigned long timeout_ms) {
18 unsigned long start = millis();
19 size_t got = 0;
20 while (got < n) {
21 int avail = Serial.available();
22 if (avail > 0) {
23 int r = Serial.readBytes(dst + got, n - got);
24 got += (r > 0 ? (size_t)r : 0);
25 continue;
26 }
27 if ((millis() - start) > timeout_ms) return false;
28 THREAD_SLEEP(1); // yield
29 }
30 return true;
31}

64// System message: instruct the Kamaji driver to do something
65// # <INSTR> <data[13]>
66// INSTR
67// - 0x00 --> reserved
68// - 0x01 --> REPORT_EN
69// - 0x02 --> VERIFY_CHECKSUM
70// - 0x03 --> FSM_SET
71// - 0x04 --> SH_BEGIN
72inline void k_handle_sys_msg(uint8_t* data, uint16_t crc, sys_flags_t* sflags) {
73 sys_instr_t instr = (sys_instr_t) data[0];
74
75 switch(instr) {
76 case (sys_instr_t::REPORT_EN):
77 {
78 ReadingDiscriminant disc = (ReadingDiscriminant) data[1];
79 uint32_t report_intv = 0;
80 memcpy(&report_intv, data + 2, sizeof(uint32_t));
81 k_enable_data_report(disc, report_intv);
82 }
83 break;
84 case (sys_instr_t::VERIFY_CHECKSUM):
85 {
86 uint32_t checksum;
87 memcpy(&checksum, data + 1, sizeof(uint32_t));
88 if(checksum != LOG_CHECKSUM) {
89 k_INVALIDCHECKSUM();
90 }
91 }
92 break;
93 case (sys_instr_t::FSM_SET):
94 {
95 FSMState new_fsm = (FSMState) data[1];
96 sflags->fsm_target = new_fsm;
97 }
98 break;
99 case (sys_instr_t::SH_BEGIN):
100 {
101 // The argument doesnt matter, once shell begins we don't care about ending it
102 sflags->sh_en = true;
103 }
104 default:
105 k_INVALIDINSTR();
106 return;
107 }
108
109 Serial.write("%OK");
110}
