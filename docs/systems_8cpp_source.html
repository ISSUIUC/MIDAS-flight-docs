34 // Reject single rogue barometer readings that are very different from the immediately prior reading

36 Barometer prev_reading;
37 constexpr float altChgThreshold = 200; // meters
38 constexpr float presChgThreshold = 500; // milibars
39 constexpr float tempChgThreshold = 10; // degrees C
40 constexpr unsigned int maxConsecutiveRejects = 3;
41 unsigned int rejects = maxConsecutiveRejects; // Always accept first reading
42 while (true) {
43 Barometer reading = arg->sensors.barometer.read();
44 bool is_rogue = std::abs(prev_reading.altitude - reading.altitude) > altChgThreshold;

47 // TODO: Log when we receive a rejection!

49 arg->rocket_data.barometer.update(prev_reading); // Reuse old reading, reject new reading

51 rejects = 0;
52 arg->rocket_data.barometer.update(reading);
53 prev_reading = reading; // Only update prev_reading with accepted readings

247void handle_tlm_command(TelemetryCommand& command, RocketSystems* arg, FSMState current_state) {
248 // maybe we should move this somewhere else but it can stay here for now
249 switch(command.command) {
250 case CommandType::RESET_KF:
251 arg->rocket_data.command_flags.should_reset_kf = true;
252 break;
253 case CommandType::SWITCH_TO_SAFE:
254 arg->rocket_data.command_flags.should_transition_safe = true;
255 break;
256 case CommandType::SWITCH_TO_PYRO_TEST:
257 arg->rocket_data.command_flags.should_transition_pyro_test = true;
258 Serial.println("Changing to pyro test");
259 break;
260 case CommandType::SWITCH_TO_IDLE:
261 arg->rocket_data.command_flags.should_transition_idle = true;
262 break;
263 case CommandType::FIRE_PYRO_A:
264 if (current_state == FSMState::STATE_PYRO_TEST) {
265 arg->rocket_data.command_flags.should_fire_pyro_a = true;
266 }
267 break;
268 case CommandType::FIRE_PYRO_B:
269 if (current_state == FSMState::STATE_PYRO_TEST) {
270 arg->rocket_data.command_flags.should_fire_pyro_b = true;
271 }
272 break;
273 case CommandType::FIRE_PYRO_C:
274 if (current_state == FSMState::STATE_PYRO_TEST) {
275 arg->rocket_data.command_flags.should_fire_pyro_c = true;
276 }
277 break;
278 case CommandType::FIRE_PYRO_D:
279 if (current_state == FSMState::STATE_PYRO_TEST) {
280 arg->rocket_data.command_flags.should_fire_pyro_d = true;
281 }
282 break;
283 case CommandType::CAM_ON:
284 arg->b2b.camera.camera_on(CAM_1);
285 arg->b2b.camera.camera_on(CAM_2);
286 arg->b2b.camera.vtx_on();
287 break;
288 case CommandType::CAM_OFF:
289 arg->b2b.camera.camera_off(CAM_1);
290 arg->b2b.camera.camera_off(CAM_2);
291 arg->b2b.camera.vtx_off();
292 break;
293 case CommandType::TOGGLE_CAM_VMUX:
294 arg->b2b.camera.vmux_toggle();
295 break;
296 default:
297 break; // how
298 }
