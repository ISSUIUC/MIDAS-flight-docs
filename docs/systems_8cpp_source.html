18void log_parse_tree(LogSink& log, size_t node_size, size_t node_count, uint8_t* data) {
19 // To recreate log formats, we need data to reconstruct the parse tree for log format
20 // This means knowing the size of each node and the pre-order traversal.
21 // We will store first the size of the node (each entry type), then a byte array of the pre order traversal.
22 // This lets us recreate the tree by converting the byte array into an array of nodes, then recreating the preorder traversal.
23 // For degenerate trees (lists), we will do the same thing, but they will be decoded as a list instead.
24 log.write_meta((uint8_t*)&node_size, sizeof(size_t)); // Store the node size
25 log.write_meta((uint8_t*)&node_count, sizeof(size_t)); // Store number of nodes
26 log.write_meta(data, node_size * node_count); // Store raw parse tree data
27}
