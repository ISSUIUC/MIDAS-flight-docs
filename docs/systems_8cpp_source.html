319void handle_tlm_command(TelemetryCommand &command, RocketSystems *arg, FSMState current_state)
320{
321 // maybe we should move this somewhere else but it can stay here for now
322 switch (command.command)
323 {
324 case CommandType::RESET_KF:
325 arg->rocket_data.command_flags.should_reset_kf = true;
326 break;
327 case CommandType::SWITCH_TO_SAFE:
328 arg->rocket_data.command_flags.should_transition_safe = true;
329 break;
330 case CommandType::SWITCH_TO_PYRO_TEST:
331 arg->rocket_data.command_flags.should_transition_pyro_test = true;
332 Serial.println("Changing to pyro test");
333 break;
334 case CommandType::SWITCH_TO_IDLE:
335 arg->rocket_data.command_flags.should_transition_idle = true;
336 break;
337 case CommandType::FIRE_PYRO_A:
338 if (current_state == FSMState::STATE_PYRO_TEST)
339 {
340 arg->rocket_data.command_flags.should_fire_pyro_a = true;
341 }
342 break;
343 case CommandType::FIRE_PYRO_B:
344 if (current_state == FSMState::STATE_PYRO_TEST)
345 {
346 arg->rocket_data.command_flags.should_fire_pyro_b = true;
347 }
348 break;
349 case CommandType::FIRE_PYRO_C:
350 if (current_state == FSMState::STATE_PYRO_TEST)
351 {
352 arg->rocket_data.command_flags.should_fire_pyro_c = true;
353 }
354 break;
355 case CommandType::FIRE_PYRO_D:
356 if (current_state == FSMState::STATE_PYRO_TEST)
357 {
358 arg->rocket_data.command_flags.should_fire_pyro_d = true;
359 }
360 break;
361 case CommandType::CAM_ON:
362 arg->b2b.camera.vtx_on(); // comes first to avoid brownouts due to high inrush current
363 arg->b2b.camera.camera_on(CAM_1);
364 arg->b2b.camera.camera_on(CAM_2);
365 break;
366 case CommandType::CAM_OFF:
367 arg->b2b.camera.vtx_off();
368 arg->b2b.camera.camera_off(CAM_1);
369 arg->b2b.camera.camera_off(CAM_2);
370 break;
371 case CommandType::TOGGLE_CAM_VMUX:
372 arg->b2b.camera.vmux_toggle();
373 break;
374 default:
375 break; // how
376 }
