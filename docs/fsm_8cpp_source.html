32template<typename T, size_t count>
33double sensor_derivative(BufferedSensorData<T, count>& sensor, double (* get_item)(T&)) {
34 auto arr = sensor.template getBufferRecent<count>();
35 auto times = sensor.template getTimesRecent<count>();
36 size_t i = 0;
37
38 double first_average = 0.0;
39 double first_average_time = 0.0;
40 for (; i < count / 2; i++) {
41 first_average += get_item(arr[i]);
42 first_average_time += pdTICKS_TO_MS(times[i]) / 1000.0;
43 }
44 first_average /= (count / 2.0);
45 first_average_time /= (count / 2.0);
46
47 double second_average = 0.0;
48 double second_average_time = 0.0;
49 for (; i < count; i++) {
50 second_average += get_item(arr[i]);
51 second_average_time += pdTICKS_TO_MS(times[i]) / 1000.0;
52 }
53 second_average /= (count / 2.0);
54 second_average_time /= (count / 2.0);
55 return (second_average - first_average) / (second_average_time - first_average_time);
56}
