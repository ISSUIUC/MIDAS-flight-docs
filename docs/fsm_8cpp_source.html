26uint32_t FSMConfiguration::calculate_crc(const FSMConfiguration& cfg) {
27 constexpr size_t FSMConfigurationCRCSize = sizeof(FSMUserThresholds) + (sizeof(FSMPyroAction) * MIDAS_NUM_PYROS) + sizeof(uint8_t);
28 uint8_t crc_buf[FSMConfigurationCRCSize];
29
30 size_t ptr = 0;
31 memcpy(crc_buf + ptr, &cfg.thresholds, sizeof(FSMUserThresholds)); ptr += sizeof(FSMUserThresholds);
32 for (int i = 0; i < MIDAS_NUM_PYROS; i++) {
33 memcpy(crc_buf + ptr, &cfg.pyro_actions[i], sizeof(FSMPyroAction)); ptr += sizeof(FSMPyroAction);

74template<typename T, size_t count>
75double sensor_derivative(BufferedSensorData<T, count>& sensor, double (* get_item)(T&)) {
76 auto arr = sensor.template getBufferRecent<count>();
77 auto times = sensor.template getTimesRecent<count>();
78 size_t i = 0;
79
80 double first_average = 0.0;
81 double first_average_time = 0.0;
82 for (; i < count / 2; i++) {
83 first_average += get_item(arr[i]);
84 first_average_time += pdTICKS_TO_MS(times[i]) / 1000.0;
85 }
86 first_average /= (count / 2.0);
87 first_average_time /= (count / 2.0);
88
89 double second_average = 0.0;
90 double second_average_time = 0.0;
91 for (; i < count; i++) {
92 second_average += get_item(arr[i]);
93 second_average_time += pdTICKS_TO_MS(times[i]) / 1000.0;
94 }
95 second_average /= (count / 2.0);
96 second_average_time /= (count / 2.0);
97 return (second_average - first_average) / (second_average_time - first_average_time);
98}

138FSMData FSM::tick_fsm(FSMTickData& fsm_data) {
139 double current_time = fsm_data.cur_time;
140 FSMData& state_data = fsm_data.cur_state;
141 FSMState& state = state_data.state;
142 uint8_t& current_motor = state_data.current_motor;
143 CommandFlags& commands = fsm_data.commands;
144 const StateEstimate& state_estimate = fsm_data.state_estimate;
145 const FSMConfiguration& config = fsm_data.fsm_config;
146 const KalmanData& kf_data = fsm_data.kf_data;
147
148 switch(state) {
149 case FSMState::STATE_SAFE:
150 // Deconflict if multiple commands are processed
151 if(commands.should_transition_safe) {
152 commands.should_transition_pyro_test = false;
153 commands.should_transition_armed = false;
154 commands.should_transition_safe = false;
155 break;
156 }
157
158 // Only switch to STATE_PYRO_TEST if triggered wirelessly
159 if(commands.should_transition_pyro_test) {
160 state = FSMState::STATE_PYRO_TEST;
161 pyro_test_entry_time = current_time;
162 commands.should_transition_pyro_test = false;
163 }
164
165 // Only switch to STATE_ARMED if triggered wirelessly.
166 if(commands.should_transition_armed) {
167 state = FSMState::STATE_ARMED;
168 current_motor = 0;
169 commands.should_transition_armed = false;
170 }
171
172 break;
173 case FSMState::STATE_PYRO_TEST:
174
175 // Force transtion to safe if requested + clear all transition flags.
176 if(commands.should_transition_safe) {
177 state = FSMState::STATE_SAFE;
178 commands.should_transition_pyro_test = false;
179 commands.should_transition_armed = false;
180 commands.should_transition_safe = false;
181 break;
182 }
183
184 // Switch back to STATE_SAFE after a certain amount of time passes
185 if((current_time - pyro_test_entry_time) > fsms_pt_disarm_t) {
186 commands.should_transition_pyro_test = false;
187 state = FSMState::STATE_SAFE;
188 }
189
190 break;
191 case FSMState::STATE_ARMED:
192
193 // Force transtion to safe if requested + clear all transition flags.
194 if(commands.should_transition_safe) {
195 state = FSMState::STATE_SAFE;
196 commands.should_transition_pyro_test = false;
197 commands.should_transition_armed = false;
198 commands.should_transition_safe = false;
199 break;
200 }
201
202 // once a significant amount of acceleration is detected change states
203 if (state_estimate.acceleration > fsms_boost_xl) {
204 launch_time = current_time; // Record launch time
205 time_entered_cur_state = current_time; // Record time for false boost detect
206 commands.FSM_should_set_cam_feed_cam1 = true; // Set camera to side cam
207 cur_state_lockin = false; // Reset "state lock in" flag for boost detect
208 state = FSMState::STATE_BOOST;
209 }
210
211 break;
212 case FSMState::STATE_BOOST:
213 // -- BEGIN FALSE MOTOR DETECTION --
214 if(!cur_state_lockin && (current_time - time_entered_cur_state) < fsms_boost_lockin_t) {
215 // The state has not been locked in yet, so we should check if we still meet the conditions
216 if(state_estimate.acceleration < fsms_boost_xl) {
217 state = (current_motor == 0) ? FSMState::STATE_ARMED : FSMState::STATE_COAST;
218 time_entered_cur_state = current_time;
219 break;
220 }
221 } else {
222 if(!cur_state_lockin) {
223 current_motor++; // When "BOOST" gets locked in, we increment the current motor.
224 cur_state_lockin = true;
225 }
226 }
227 // -- END FALSE MOTOR DETECTION --
228
229
230 // Burnout detection
231 if (state_estimate.acceleration < fsms_burnout_xl) {
232 time_entered_cur_state = current_time;
233 state = FSMState::STATE_COAST;
234 cur_state_lockin = false; // Reset "state lock in" flag for false burnout detection
235 }
236 break;
237 case FSMState::STATE_COAST: {
238 // if low acceleration is too brief than go on to the previous state
239
240 // -- BEGIN FALSE BURNOUT DETECTION --
241 if(!cur_state_lockin && (current_time - time_entered_cur_state) < fsms_burnout_lockin_t) {
242 // The state has not been locked in yet, so we should check if we still meet the conditions
243 if(state_estimate.acceleration >= fsms_burnout_xl) {
244 state = FSMState::STATE_BOOST;
245 // Note: We do not reset the "time since entering current state" timer, as we're assuming we're still in the previous boost phase.
246 cur_state_lockin = true; // We are going back to a previously "locked-in" state.
247 break;
248 }
249 } else {
250 cur_state_lockin = true;
251 }
252 // -- END FALSE BURNOUT DETECTION --
253
254 // -- BEGIN NEXT STAGE IGNITION DETECT --
255 // After burnout is confirmed, check for another motor ignition (multistage)
256 if(cur_state_lockin && state_estimate.acceleration > fsms_boost_xl) {
257 time_entered_cur_state = current_time;
258 cur_state_lockin = false;
259 apogee_detect_start = 0; // Reset apogee detection
260 state = FSMState::STATE_BOOST;
261 break;
262 }
263 // -- END NEXT STAGE IGNITION DETECT --
264
265 // -- BEGIN APOGEE DETECT --
266 // Condition 1: Vertical speed low
267 bool apog_detect_low_speed = (state_estimate.vertical_speed <= fsms_apogee_detect_spd);
268
269 // Condition 2: Cruise lockout
270 bool apog_detect_cruise_lockout = (!config.thresholds.cruise_lockout_en || kf_data.velocity.vx <= fsms_cruise_lockout_spd);
271 // Note: This evaluates to TRUE (no lockout) if the lockout is disabled, OR if the condition is met
272
273 if (apog_detect_low_speed && apog_detect_cruise_lockout) {
274 // Start the consecutive timer if not already running
275 if (apogee_detect_start == 0) {
276 apogee_detect_start = current_time;
277 }
278
279 // Transition to DROGUE only after conditions are met for fsms_apogee_lockin_t consecutive ms
280 if (current_time - apogee_detect_start >= fsms_apogee_lockin_t) {
281 time_entered_cur_state = current_time;
282 apogee_time = current_time;
283 state = FSMState::STATE_DROGUE;
284 apogee_detect_start = 0;
285
286 commands.FSM_should_swap_camera_feed = true;
287 }
288 } else {
289 // Conditions not met: reset the timer
290 apogee_detect_start = 0;
291 }
292 // -- END APOGEE DETECT --
293 break;
294 }
295
296 case FSMState::STATE_DROGUE:
297 // Simple: If we hit the main threshold, deploy main.
298 // Make sure to respect the MAIN_LOCKOUT_T rule.
299
300 if(state_estimate.altitude <= config.thresholds.main_alt && (current_time - time_entered_cur_state) > fsms_main_lockout_t) {
301 time_entered_cur_state = current_time;
302 state = FSMState::STATE_MAIN;
303 }
304 break;
305
306 case FSMState::STATE_MAIN: {
307 bool landed_speed = (abs(state_estimate.vertical_speed) <= fsms_landed_detect_spd);
308 bool landed_lockout_passed = (current_time - apogee_time) > fsms_landed_t_lockout;
309
310 if (landed_speed && landed_lockout_passed) {
311 if (landed_detect_start == 0) {
312 landed_detect_start = current_time;
313 }
314
315 if (current_time - landed_detect_start >= fsms_landed_entry_t) {
316 time_entered_cur_state = current_time;
317 cur_state_lockin = false;
318 state = FSMState::STATE_LANDED;
319 landed_detect_start = 0;
320 }
321 } else {
322 landed_detect_start = 0;
323 }
324 break;
325 }
326
327 case FSMState::STATE_LANDED:
328 // Landing lock-in
329 if((current_time - time_entered_cur_state) > fsms_landed_t) {
330
331 if(!cur_state_lockin) {
332 cur_state_lockin = true;
333 commands.FSM_should_power_save = true;
334 }
335
336 // Check for any telem transitions
337 // Force transtion to safe if requested + clear all transition flags.
338 if(commands.should_transition_safe) {
339 state = FSMState::STATE_SAFE;
340 commands.should_transition_pyro_test = false;
341 commands.should_transition_armed = false;
342 commands.should_transition_safe = false;
343 }
344
345 break;
346 }
347
348 // If we still detect falling movement, go back to MAIN.
349 if (abs(state_estimate.vertical_speed) > fsms_landed_detect_spd) {
350 state = FSMState::STATE_MAIN;
351 time_entered_cur_state = current_time;
352 }
353
354 break;
355 }
356
357 return state_data;
358}
