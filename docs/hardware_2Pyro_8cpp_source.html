87PyroState Pyro::tick(FSMState fsm_state, Orientation orientation, CommandFlags& telem_commands) {
88 PyroState new_pyro_state = PyroState();
89 double current_time = pdTICKS_TO_MS(xTaskGetTickCount());
90
91 if (fsm_state == FSMState::STATE_SAFE) {
92 disarm_all_channels(new_pyro_state);
93 return new_pyro_state;
94 }
95
96 // If the state is not SAFE, we arm the global arm pin
97 new_pyro_state.is_global_armed = true;
98 gpioDigitalWrite(PYRO_GLOBAL_ARM_PIN, HIGH);
99
100 switch (fsm_state) {
101 case FSMState::STATE_IDLE:
102 reset_pyro_safety(); // Ensure that pyros can be fired when we transition away from this state
103 break;
104 case FSMState::STATE_PYRO_TEST:
105
106 if(safety_has_fired_pyros_this_cycle) {
107 // If a fire pyro command has already be acknowledged, do not acknowledge more commands, just fire pyro for the defined time
108 // then, transition to SAFE.
109 if((current_time - safety_pyro_start_firing_time) >= PYRO_TEST_FIRE_TIME) {
110 telem_commands.should_transition_safe = true;
111 disarm_all_channels(new_pyro_state);
112 telem_commands.should_fire_pyro_a = false;
113 telem_commands.should_fire_pyro_b = false;
114 telem_commands.should_fire_pyro_c = false;
115 telem_commands.should_fire_pyro_d = false;
116
117 reset_pyro_safety();
118 }
119 break;
120 }
121
122 // Respond to telem commands to fire igniters
123 if(telem_commands.should_fire_pyro_a) {
124 // Fire pyro channel "A"
125 new_pyro_state.channel_firing[0] = true;
126 gpioDigitalWrite(PYROA_FIRE_PIN, HIGH);
127 set_pyro_safety();
128 }
129
130 if(telem_commands.should_fire_pyro_b) {
131 // Fire pyro channel "B"
132 new_pyro_state.channel_firing[1] = true;
133 gpioDigitalWrite(PYROB_FIRE_PIN, HIGH);
134
135 set_pyro_safety();
136 }
137
138 if(telem_commands.should_fire_pyro_c) {
139 // Fire pyro channel "C"
140 new_pyro_state.channel_firing[2] = true;
141 gpioDigitalWrite(PYROC_FIRE_PIN, HIGH);
142
143 set_pyro_safety();
144 }
145
146 if(telem_commands.should_fire_pyro_d) {
147 // Fire pyro channel "D"
148 new_pyro_state.channel_firing[0] = true;
149 gpioDigitalWrite(PYROD_FIRE_PIN, HIGH);
150
151 set_pyro_safety();
152 }
153
154 break;
155 case FSMState::STATE_SUSTAINER_IGNITION:
156 // Fire "Pyro C" to ignite sustainer (Pyro C is motor channel)
157 // Additionally, check if orientation allows for firing
158 if (can_fire_igniter(orientation)) {
159 new_pyro_state.channel_firing[2] = true;
160 gpioDigitalWrite(PYROC_FIRE_PIN, HIGH);
161 }
162 break;
163 case FSMState::STATE_DROGUE_DEPLOY:
164 // Fire "Pyro A" to deploy upper stage drogue
165 new_pyro_state.channel_firing[0] = true;
166 gpioDigitalWrite(PYROA_FIRE_PIN, HIGH);
167 break;
168 case FSMState::STATE_MAIN_DEPLOY:
169 // Fire "Pyro B" to deploy main.
170 new_pyro_state.channel_firing[1] = true;
171 gpioDigitalWrite(PYROB_FIRE_PIN, HIGH);
172 break;
173 default:
174 break;
175 }
176
177 return new_pyro_state;
178}
