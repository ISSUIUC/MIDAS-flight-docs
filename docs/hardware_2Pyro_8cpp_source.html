67PyroState Pyro::tick(FSMState fsm_state, Orientation orientation) {
68 PyroState new_pyro_state = PyroState();
69
70 // If the state is IDLE or any state after that, we arm the global arm pin
71 new_pyro_state.is_global_armed = true;
72 gpioDigitalWrite(PYRO_GLOBAL_ARM_PIN, HIGH);
73
74 switch (fsm_state) {
75 case FSMState::STATE_SUSTAINER_IGNITION:
76 // Fire "Pyro A" to ignite sustainer
77 // Additionally, check if orientation allows for firing
78 if (can_fire_igniter(orientation)) {
79 new_pyro_state.channels[0].is_armed = true;
80 new_pyro_state.channels[0].is_firing = true;
81 gpioDigitalWrite(PYROA_ARM_PIN, HIGH);
82 gpioDigitalWrite(PYROA_FIRE_PIN, HIGH);
83 }
84 break;
85 case FSMState::STATE_DROGUE_DEPLOY:
86 // Fire "Pyro B" to deploy upper stage drogue
87 new_pyro_state.channels[1].is_armed = true;
88 new_pyro_state.channels[1].is_firing = true;
89 gpioDigitalWrite(PYROB_ARM_PIN, HIGH);
90 gpioDigitalWrite(PYROB_FIRE_PIN, HIGH);
91 break;
92 case FSMState::STATE_MAIN_DEPLOY:
93 // Fire "Pyro C" to deploy main.
94 new_pyro_state.channels[2].is_armed = true;
95 new_pyro_state.channels[2].is_firing = true;
96 gpioDigitalWrite(PYROC_ARM_PIN, HIGH);
97 gpioDigitalWrite(PYROC_FIRE_PIN, HIGH);
98 break;
99 default:
100 break;
101 }
102
103 return new_pyro_state;
104}
