39Vec3 quaternionToEuler(float qr, float qi, float qj, float qk, bool degrees) {
40 float sqr = sq(qr);
41 float sqi = sq(qi);
42 float sqj = sq(qj);
43 float sqk = sq(qk);
44
45 Vec3 euler;
46 euler.x = atan2(2.0 * (qi * qj + qk * qr), (sqi - sqj - sqk + sqr)); // roll
47 euler.y = asin(-2.0 * (qi * qk - qj * qr) / (sqi + sqj + sqk + sqr)); // yaw
