66Vec3 quaternionToEuler(float qr, float qi, float qj, float qk, bool degrees)
67{
68 float sqr = sq(qr);
69 float sqi = sq(qi);
70 float sqj = sq(qj);
71 float sqk = sq(qk);
72
73 Vec3 euler;
74 euler.x = atan2(2.0 * (qi * qj + qk * qr), (sqi - sqj - sqk + sqr)); // roll
75 euler.y = asin(-2.0 * (qi * qk - qj * qr) / (sqi + sqj + sqk + sqr)); // yaw
76 euler.z = -1 * atan2(2.0 * (qj * qk + qi * qr), (-sqi - sqj + sqk + sqr)); // pitch
77 return euler;
78}

116float angular_difference(float pitch1, float yaw1, float pitch2, float yaw2)
117{
118 auto [x1, y1, z1] = euler_to_vector(pitch1, yaw1);
119 auto [x2, y2, z2] = euler_to_vector(pitch2, yaw2);
120
121 float mag1 = sqrt(x1 * x1 + y1 * y1 + z1 * z1);
122 float mag2 = sqrt(x2 * x2 + y2 * y2 + z2 * z2);
123
124 float dot_product = x1 * x2 + y1 * y2 + z1 * z2;
125
126 return std::acos(dot_product / (mag1 * mag2));
127}

133 float roll = rpy_vec.x;
134 float pitch = rpy_vec.y;
