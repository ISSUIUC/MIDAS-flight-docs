525 thrust_curve = &motor_data.at("Sustainer"); // Sustainer
526 else
527 {
528 thrust_out.setZero();
529 return; // No thrust before ignition
530 }
531
532 // Handle case where timestamp is before or after available data
533 if (timestamp <= thrust_curve->begin()->first)
534 {
535 thrust_out = Eigen::Vector3f(thrust_curve->begin()->second, 0.f, 0.f);
536 }
537 else if (timestamp >= thrust_curve->rbegin()->first)
538 {
539 thrust_out.setZero(); // assume motor burned out after curve ends
540 }
541 else
542 {
543 // Find interpolation interval
544 auto it_upper = thrust_curve->lower_bound(timestamp);
545 auto it_lower = std::prev(it_upper);
546
547 float x0 = it_lower->first;
548 float y0 = it_lower->second;
549 float x1 = it_upper->first;
550 float y1 = it_upper->second;
551
552 float interpolated_thrust = linearInterpolation(x0, y0, x1, y1, timestamp);
553 thrust_out = Eigen::Vector3f(interpolated_thrust, 0.f, 0.f);
554 }

557 // BodyToGlobal(angles, thrust_out);
