179 index = std::clamp(index, 0, (int)AERO_DATA_SIZE - 1);

181 Ca = aero_data[index].CA_power_on;

183 // aerodynamic force
184 // Body frame
185 float Fax = 0; // instead of mag square --> mag * vel_x
186 if ((fsm > FSMState::STATE_IDLE))
187 {
188 Fax = -0.5 * rho * (vel_magnitude_ms) * float(Ca) * (pi * r * r) * x_k(1, 0);
189 }
190 float Fay = 0; // assuming no aerodynamic effects
191 float Faz = 0; // assuming no aerodynamic effects

193 // acceleration due to gravity
194 float gx = g_global(0, 0);
195 float gy = g_global(1, 0);
196 float gz = g_global(2, 0);

205 float Ftz = Ft_body(2, 0);

214
215 Eigen::Matrix<float, 3, 1> v_dot; // we compute everything in the body frame for accelerations, and then convert those accelerations to global frame
216 v_dot << ((Fax + Ftx) / curr_mass_kg - (omega_rps.vy * vz_body - omega_rps.vz * vy_body) + x_k(2, 0)) / 2,
217 ((Fay + Fty) / curr_mass_kg - (omega_rps.vz * vx_body - omega_rps.vx * vz_body) + x_k(5, 0) / 2),
218 ((Faz + Ftz) / curr_mass_kg - (omega_rps.vx * vy_body - omega_rps.vy * vx_body) + x_k(8, 0) / 2);
219
220 BodyToGlobal(angles_rad, v_dot);
221
222 xdot << x_k(1, 0), v_dot(0, 0) + gx,
223 0.0,
224
225 x_k(4, 0), v_dot(1, 0) + gy,
226 0.0,
227
228 x_k(7, 0), v_dot(2, 0) + gz,
229 0.0;
230
231 // priori step
232 x_priori = (xdot * dt) + x_k;
233
234 float coeff = 0;
235 if ((fsm > FSMState::STATE_IDLE))
236 {
237 coeff = -pi * Ca * (r * r) * rho / curr_mass_kg;
238 }
239
240 setF(dt, omega_rps.vx, omega_rps.vy, omega_rps.vz, coeff, vx_body, vy_body, vz_body);
241
242 P_priori = (F_mat * P_k * F_mat.transpose()) + Q;

530 float y1 = it_upper->second;
531

533 thrust_out = Eigen::Vector3f(interpolated_thrust, 0.f, 0.f);
