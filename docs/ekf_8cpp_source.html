27void EKF::initialize(RocketSystems *args)
28{
29 // Orientation orientation = args->rocket_data.orientation.getRecentUnsync();
30 float sum = 0;
31
32 for (int i = 0; i < 30; i++)
33 {
34 Barometer barometer = args->rocket_data.barometer.getRecent();
35 sum += barometer.altitude;
36 }
37
38 // set x_k - 6 states: [x, vx, y, vy, z, vz]
39 x_k.setZero();
40 x_k(0, 0) = sum / 30; // initial altitude (x position)
41 x_k(2, 0) = 0; // y position
42 x_k(4, 0) = 0; // z position
43
44 F_mat.setZero(); // Initialize with zeros
45 B_mat.setZero(); // Initialize control input matrix
46
47 setQ(s_dt, spectral_density_);
48 H.setZero();
49 H(0, 0) = 1; // barometer measures x position (altitude)
50 H(1, 0) = 1; // GPS measures x position (altitude)
51 H(2, 2) = 1; // GPS measures y position (east)
52 H(3, 4) = 1; // GPS measures z position (north)
53
54 P_k.setZero();
55 P_k.block<2, 2>(0, 0) = Eigen::Matrix2f::Identity() * 1e-2f; // x block (pos,vel)
56 P_k.block<2, 2>(2, 2) = Eigen::Matrix2f::Identity() * 1e-2f; // y block (pos,vel)
57 P_k.block<2, 2>(4, 4) = Eigen::Matrix2f::Identity() * 1e-2f; // z block (pos,vel)
58
59 // set Measurement Noise Matrix
60 R(0, 0) = 1.9; // barometer noise
61 R(1, 1) = 4.0f; // GPS altitude noise (lower trust)
62 R(2, 2) = 3.0f; // GPS east noise
63 R(3, 3) = 3.0f; // GPS north noise
64}

74void EKF::priori(float dt, AngularKalmanData &orientation, FSMState fsm, Acceleration acceleration)
75{
76 setF(dt);
77 setB(dt);
78 // Compute control input at current time step
79 Eigen::Matrix<float, 3, 1> sensor_accel_global_g = Eigen::Matrix<float, 3, 1>::Zero();
80 sensor_accel_global_g(0, 0) = acceleration.ax + 0.045f;
81 sensor_accel_global_g(1, 0) = acceleration.ay - 0.065f;
82 sensor_accel_global_g(2, 0) = acceleration.az - 0.06f;
83 euler_t angles_rad = orientation.getEuler();
84 BodyToGlobal(angles_rad, sensor_accel_global_g);
85 // Do not apply acceleration if on pad
86 float g_ms2 = (fsm > FSMState::STATE_IDLE) ? gravity_ms2 : 0.0f;
87 u_control(0, 0) = sensor_accel_global_g(0, 0) * g_ms2;
88 u_control(1, 0) = sensor_accel_global_g(1, 0) * g_ms2;
89 u_control(2, 0) = sensor_accel_global_g(2, 0) * g_ms2;
90 // Predict state at time t and covariance
91 x_priori = F_mat * x_k + B_mat * u_control;
92 P_priori = (F_mat * P_k * F_mat.transpose()) + Q;
93}

102void EKF::update(Barometer barometer, Acceleration acceleration, AngularKalmanData orientation, FSMState FSM_state, GPS &gps)
103{
104 // if on pad take last 10 barometer measurements for init state
105 if (FSM_state == FSMState::STATE_IDLE)
106 {
107 float sum = 0;
108 float data[10];
109 alt_buffer.readSlice(data, 0, 10);
110 for (float i : data)
111 {
112 sum += i;
113 }
114 KalmanState kalman_state = (KalmanState){sum / 10.0f, 0, 0, 0, 0, 0, 0, 0, 0};
115 setState(kalman_state);
116 }
117
118 // Update GPS reference point if in IDLE state (needed before GPS measurement update)
119 reference_GPS(gps, FSM_state);
120
121 // Check if GPS has valid fix
122 bool gps_available = (gps.fix_type != 0 && gps.latitude != 0 && gps.longitude != 0);
123
124 // Build measurement vector and select appropriate H and R matrices
125 Eigen::Matrix<float, NUM_STATES, NUM_STATES> identity = Eigen::Matrix<float, NUM_STATES, NUM_STATES>::Identity();
126
127 if (gps_available)
128 {
129 // GPS available: use full H matrix
130 float lat = gps.latitude / 1e7; // dividing by 1e7 to convert from int to float
131 float lon = gps.longitude / 1e7;
132 float alt = gps.altitude;
133 // Convert GPS to ECEF
134 std::vector<float> rocket_cords = gps_to_ecef(lat, lon, alt);
135 std::vector<float> reference_cord = gps_to_ecef(gps_latitude_original, gps_longitude_original, 0);
136 float gps_latitude_original_rad = gps_latitude_original * pi / 180;
137 float gps_longitude_original_rad = gps_longitude_original * pi / 180;
138 double dx = rocket_cords[0] - reference_cord[0];
139 double dy = rocket_cords[1] - reference_cord[1];
140 double dz = rocket_cords[2] - reference_cord[2];
141 float east = -std::sin(gps_longitude_original_rad) * dx + std::cos(gps_longitude_original_rad) * dy;
142 float north = -std::sin(gps_latitude_original_rad) * std::cos(gps_longitude_original_rad) * dx - std::sin(gps_latitude_original_rad) * std::sin(gps_longitude_original_rad) * dy + std::cos(gps_latitude_original_rad) * dz;
143 // Build measurement vector
144 Eigen::Matrix<float, 4, 1> y_combined;
145 y_combined(0) = barometer.altitude; // barometer altitude
146 y_combined(1) = alt; // GPS altitude
147 y_combined(2) = east; // GPS east
148 y_combined(3) = north; // GPS north
149 // Innovation
150 Eigen::Matrix<float, 4, 1> innovation = y_combined - H * x_priori;
151 // Innovation covariance
152 Eigen::Matrix<float, 4, 4> S = H * P_priori * H.transpose() + R;
153 // Kalman gain
154 Eigen::Matrix<float, NUM_STATES, 4> K = P_priori * H.transpose() * S.inverse();
155 // Update state and covariance
156 x_k = x_priori + K * innovation;
157 P_k = (identity - K * H) * P_priori;
158
159 gps_latitude_last = lat;
160 gps_longitude_last = lon;
161 }
162 else
163 {
164 // GPS not available, use only barometer H matrix
165 Eigen::Matrix<float, 1, NUM_STATES> H_baro = H.block<1, NUM_STATES>(0, 0);
166 Eigen::Matrix<float, 1, 1> R_baro;
167 R_baro(0, 0) = R(0, 0);
168 // Build measurement vector: [baro_alt]
169 Eigen::Matrix<float, 1, 1> y_baro;
170 y_baro(0) = barometer.altitude;
171
172 // Innovation
173 Eigen::Matrix<float, 1, 1> innovation = y_baro - H_baro * x_priori;
174 // Innovation covariance
175 Eigen::Matrix<float, 1, 1> S = H_baro * P_priori * H_baro.transpose() + R_baro;
176 // Kalman gain
177 Eigen::Matrix<float, NUM_STATES, 1> K = P_priori * H_baro.transpose() / S(0, 0);
178 // Update state and covariance
179 x_k = x_priori + K * innovation;
180 P_k = (identity - K * H_baro) * P_priori;
181 }
182
183 bool is_landed = (FSM_state == FSMState::STATE_LANDED);
184 if (is_landed)
185 {
186 if (was_landed_last)
187 {
188 landed_state_duration += s_dt; // Accumulate time in LANDED state
189 }
190 else
191 {
192 landed_state_duration = s_dt; // Reset timer
193 }
194
195 // Only reset velocities if we've been landed for at least 0.5 seconds
196 if (landed_state_duration >= 0.5f)
197 {
198 x_k(3, 0) = 0.0f; // velocity y (vy) = 0 when landed
199 x_k(5, 0) = 0.0f; // velocity z (vz) = 0 when landed
200 }
201 }
202 else
203 {
204 // Negate velocity z if it's negative (so negative becomes positive)
205 if (x_k(5, 0) < 0) // velocity z (vz)
206 {
207 x_k(5, 0) = -x_k(5, 0); // Negate negative velocity z to make it positive
208 }
209 }
210
211 // Update output state structure
212 kalman_state.state_est_pos_x = x_k(0, 0);
213 kalman_state.state_est_vel_x = x_k(1, 0);
214 kalman_state.state_est_accel_x = u_control(0, 0);
215 kalman_state.state_est_pos_y = x_k(2, 0);
216 kalman_state.state_est_vel_y = x_k(3, 0);
217 kalman_state.state_est_accel_y = u_control(1, 0);
218 kalman_state.state_est_pos_z = x_k(4, 0);
219 kalman_state.state_est_vel_z = x_k(5, 0);
220 kalman_state.state_est_accel_z = u_control(2, 0);
221 state.position = (Position){kalman_state.state_est_pos_x, kalman_state.state_est_pos_y, kalman_state.state_est_pos_z};
222 state.velocity = (Velocity){kalman_state.state_est_vel_x, kalman_state.state_est_vel_y, kalman_state.state_est_vel_z};
223 state.acceleration = (Acceleration){kalman_state.state_est_accel_x, kalman_state.state_est_accel_y, kalman_state.state_est_accel_z};
224}

298void EKF::setQ(float dt, float sd)
299{
300 // discrete-time integrated acceleration noise
301 float sigma_a = 0.2f;
302 Q.setZero();
303 // X axis
304 Q(0,0) = pow(dt,4)/4.0f * sigma_a*sigma_a;
305 Q(0,1) = pow(dt,3)/2.0f * sigma_a*sigma_a;
306 Q(1,0) = Q(0,1);
307 Q(1,1) = pow(dt,2) * sigma_a*sigma_a;
308 // Y axis
309 Q(2,2) = pow(dt,4)/4.0f * sigma_a*sigma_a;
310 Q(2,3) = pow(dt,3)/2.0f * sigma_a*sigma_a;
311 Q(3,2) = Q(2,3);
312 Q(3,3) = pow(dt,2) * sigma_a*sigma_a;
313 // Z axis
314 Q(4,4) = pow(dt,4)/4.0f * sigma_a*sigma_a;
315 Q(4,5) = pow(dt,3)/2.0f * sigma_a*sigma_a;
316 Q(5,4) = Q(4,5);
317 Q(5,5) = pow(dt,2) * sigma_a*sigma_a;
318}
