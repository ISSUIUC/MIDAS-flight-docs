18template <typename Angles>
19void BodyToGlobal(Angles &angles_rad, Eigen::Matrix<float, 3, 1> &body_vec)
20{
21 Eigen::Matrix3f roll, pitch, yaw;
22 roll << 1., 0., 0.,
23 0., cos(angles_rad.roll), -sin(angles_rad.roll),
24 0., sin(angles_rad.roll), cos(angles_rad.roll);
25
26 pitch << cos(angles_rad.pitch), 0., sin(angles_rad.pitch),
27 0., 1., 0.,
28 -sin(angles_rad.pitch), 0., cos(angles_rad.pitch);
29
30 yaw << cos(angles_rad.yaw), -sin(angles_rad.yaw), 0.,
31 sin(angles_rad.yaw), cos(angles_rad.yaw), 0.,
32 0., 0., 1.;
33
34 Eigen::Matrix3f rotation_matrix = yaw * pitch * roll;
35 // temp = body_vec expressed in global frame (rotation_matrix * body_vec)
36 Eigen::Vector3f temp = rotation_matrix * body_vec;
37 body_vec(0, 0) = temp(0);
38 body_vec(1, 0) = temp(1);
39 body_vec(2, 0) = temp(2);
40}

50template <typename Angles>
51void GlobalToBody(Angles &angles_rad, Eigen::Matrix<float, 3, 1> &global_vec)
52{
53 Eigen::Matrix3f roll, pitch, yaw;
54
55 roll << cos(angles_rad.roll), -sin(angles_rad.roll), 0.,
56 sin(angles_rad.roll), cos(angles_rad.roll), 0.,
57 0., 0., 1.;
58
59 // Pitch about Y (tilt forward/back)
60 pitch << cos(angles_rad.pitch), 0., sin(angles_rad.pitch),
61 0., 1., 0.,
62 -sin(angles_rad.pitch), 0., cos(angles_rad.pitch);
63
64 // Yaw about X (turn around up axis)
65 yaw << 1., 0., 0.,
66 0., cos(angles_rad.yaw), -sin(angles_rad.yaw),
67 0., sin(angles_rad.yaw), cos(angles_rad.yaw);
68
69 Eigen::Matrix3f rotation_matrix = yaw * pitch * roll;
70 Eigen::Vector3f temp = rotation_matrix.transpose() * global_vec;
71 Eigen::Matrix3f R_zup_to_xup;
72 R_zup_to_xup << 1, 0, 0,
73 0, 1, 0,
74 0, 0, 1;
75
76 global_vec = (R_zup_to_xup * rotation_matrix).transpose() * global_vec;
77}

83inline std::vector<float> gps_to_ecef(float lat, float lon, float alt) {
84 // Convert to radians
85 lat *= pi / 180.0;
86 lon *= pi / 180.0;
87
88 double N = A / std::sqrt(1 - E_SQ * std::sin(lat) * std::sin(lat));
89
90 float ecef_x = (N + alt) * std::cos(lat) * std::cos(lon);
91 float ecef_y = (N + alt) * std::cos(lat) * std::sin(lon);
92 float ecef_z = ((1 - E_SQ) * N + alt) * std::sin(lat);
93
94 return {ecef_x, ecef_y, ecef_z};
95}

101inline std::vector<float> ecef_to_enu(std::vector<float> curr_ecef, std::vector<float> ref_ecef, std::vector<float> ref_gps) {
102 float ref_lat = ref_gps[0] * pi / 180.0;
103 float ref_lon = ref_gps[1] * pi / 180.0;
104
105 float dx = curr_ecef[0] - ref_ecef[0];
106 float dy = curr_ecef[1] - ref_ecef[1];
107 float dz = curr_ecef[2] - ref_ecef[2];
108
109 float east = - std::sin(ref_lon) * dx
110 + std::cos(ref_lon) * dy;
111 float north = - std::sin(ref_lat) * std::cos(ref_lon) * dx
112 - std::sin(ref_lat) * std::sin(ref_lon) * dy
113 + std::cos(ref_lat) * dz;
114 float up = std::cos(ref_lat) * std::cos(ref_lon) * dx
115 + std::cos(ref_lat) * std::sin(ref_lon) * dy
116 + std::sin(ref_lat) * dz;
117
118 return {east, north, up};
119}
